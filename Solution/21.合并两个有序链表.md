# 21. 合并两个有序链表

[链接](https://leetcode-cn.com/problems/merge-two-sorted-lists/description/)

将两个有序链表合并为一个新的有序链表并返回。新链表是通过拼接给定的两个链表的所有节点组成的。 

**示例：**

```
输入：1->2->4, 1->3->4
输出：1->1->2->3->4->4
```

**我的实现**

```c++
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode* mergeTwoLists(ListNode* l1, ListNode* l2) {
        if( l1 == nullptr )
            return l2;
        if( l2 == nullptr )
            return l1;
        
        ListNode* pHead = nullptr;
        ListNode* pCurrent = nullptr;
        
        while( l1 != nullptr && l2 != nullptr )
        {
            if( l1->val < l2->val )
            {
                if( pHead == nullptr )
                {
                    pHead = l1;
                    pCurrent = l1;
                }else
                {
                    pCurrent->next = l1;
                    pCurrent = pCurrent->next;
                }
                
                l1 = l1->next;
            }else
            {
                if( pHead == nullptr )
                {
                    pHead = l2;
                    pCurrent = l2;
                }else
                {
                    pCurrent->next = l2;
                    pCurrent = pCurrent->next;
                }
                
                l2 = l2->next;
            }
        }
        if( l1 != nullptr )
            pCurrent->next = l1;
        if( l2 != nullptr )
            pCurrent->next = l2;
            
        return pHead;
    }
};
```
